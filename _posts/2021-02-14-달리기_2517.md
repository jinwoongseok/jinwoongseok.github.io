[문제링크](https://www.acmicpc.net/problem/1761)
### [문제풀이]
세그먼트 트리 풀이와 머지소트를 응용한 풀이 2개로 풀 수 있다.  
세그먼트 트리로 풀이할 때 생각해주어야 하는 점은 능력의 범위가 매우크므로 이것을 배열로 만들어주기 위해 압축을 시켜야한다는 점이다(능력은 전부 다른 값)  
예시) 200 10 20 -> 3 1 2 로 표현 가능

### [소스코드]
- 세그먼트 트리를 이용한 풀이
~~~cpp
#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;
#define endl "\n"
#define MAX 500001

int n;
struct player{
    int ability;
    int idx;
};

int segtree[MAX*4];
player arr[MAX];

bool compa(player a, player b){
    return a.ability < b.ability;
}

bool compi(player a, player b){
    return a.idx < b.idx;
}

void update_tree(int node, int l, int r, int ab){
    if(l>ab || r<ab)return;
    if(l==r){
        segtree[node]++;
        return;
    }
    segtree[node]++;
    int mid=(l+r)/2;
    update_tree(node*2,l,mid,ab);
    update_tree(node*2+1,mid+1,r,ab);
}

int query(int node, int l, int r, int s, int e){
    if(s<=l && r<=e)return segtree[node];
    if(r<s || e<l)return 0;
    int mid=(l+r)/2;
    int leftSide=query(node*2, l, mid, s, e);
    int rightSide=query(node*2+1, mid+1, r, s, e);
    return leftSide+rightSide;
}

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    cin >> n;
    for(int i=1;i<=n;i++){
        int a;
        cin >> a;
        player p;
        p.ability=a;
        p.idx=i;
        arr[i]=p;
    }
    sort(arr+1,arr+n+1,compa);
    //능력 압축
    for(int i=1;i<=n;i++){
        arr[i].ability=i;
    }
    //인덱스 순으로 다시 정렬
    sort(arr+1,arr+n+1,compi);
    for(int i=1;i<=n;i++){
        int cnt = query(1, 1, n, 1, arr[i].ability);
        cout << i-cnt << endl;
        update_tree(1,1,n,arr[i].ability);
    }
    return 0;
}
~~~

- 병합정렬을 이용한 풀이
~~~cpp
#include <iostream>
#include <algorithm>
#include <vector>
#include <cstring>
#define endl "\n"
using namespace std;

struct runner {
	int value;
	int position;
};

int n;
int ranks[500000];
vector<runner> player;

void merge(int s, int mid, int e) {
	int p1 = s;
	int p2 = mid + 1;
	int k = s;
	int vtemp[500000] = { 0 };
	int ptemp[500000] = { 0 };
	while (p1 <= mid && p2 <= e) {
		if (player[p1].value >= player[p2].value) {
			ptemp[k] = player[p1].position;
			vtemp[k++] = player[p1++].value;
		}
		else {
			int count = mid - p1 + 1;
			ranks[player[p2].position] -= count;
			ptemp[k] = player[p2].position;
			vtemp[k++] = player[p2++].value;
		}
	}
	while (p1 <= mid) {
		ptemp[k] = player[p1].position;
		vtemp[k++] = player[p1++].value;
	}
	while (p2 <= e) {
		ptemp[k] = player[p2].position;
		vtemp[k++] = player[p2++].value;
	}
	for (int i = s; i <= e; i++) {
		player[i].value = vtemp[i];
		player[i].position = ptemp[i];
	}

}

void mergeSort(int s, int e) {
	if (s < e) {
		int mid = (s + e) / 2;
		mergeSort(s, mid);
		mergeSort(mid + 1, e);
		merge(s, mid, e);
	}
}

int main() {
	ios::sync_with_stdio(false);
	cin.tie(NULL);
	cout.tie(NULL);
	cin >> n;
	for (int i = 0; i < n; i++) {
		int val;
		cin >> val;
		runner r;
		r.position = i;
		r.value = val;
		ranks[i] = i;
		player.push_back(r);
	}
	mergeSort(0, n - 1);
	for (int i = 0; i < n; i++) {
		cout << ranks[i] + 1 << "  ";
	}
	return 0;
}
~~~
