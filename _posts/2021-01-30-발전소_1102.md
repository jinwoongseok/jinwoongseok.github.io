[문제링크](https://www.acmicpc.net/problem/1102)
### [문제풀이]
비트마스킹을 이용한 다이나믹 프로그래밍이다. 현재 상태를 dp 배열 인자로 사용해 각 상태에서의 최소 비용을 찾아내는 방식이다.  
팁으로 비트마스킹을 활용할 때, 1~N 까지의 범위설정보단 0~N-1 범위설정이 편하다.  

### [소스코드]
~~~cpp
#include <iostream>
#include <algorithm>
#include <cstring>
#include <cmath>
#define MAX 16
#define endl "\n"
using namespace std;

int w[MAX][MAX];
int dp[1 << 16];
int n;
bool onoff[16];
string s;
int p;
const int INF = 987654321;

int bit_set(int orgbit, int pos) {
	return orgbit | (1 << pos);
}

bool bit_check(int orgbit, int pos) {
	return orgbit & (1 << pos);
}

int bit_count(int x) {
	int count = 0;
	while (x) {
		if (x & 1)count++;
		x /= 2;
	}
	return count;
}

int main() {
	ios::sync_with_stdio(false);
	cin.tie(NULL);
	cout.tie(NULL);
	cin >> n;
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < n; j++) {
			cin >> w[i][j];
		}
	}
	for (int i = 0; i < (1 << n); i++) {
		dp[i] = INF;
	}
	cin >> s;
	for (int i = 0; i < s.size(); i++) {
		if (s[i] == 'Y')onoff[i] = 1;
	}
	cin >> p;
	int state = 0;
	for (int i = 0; i < n; i++) {
		if (onoff[i] == true) {
			state = bit_set(state, i);
		}
	}

	dp[state] = 0;
	for (int state = 0; state < (1 << n); state++) {
		for (int i = 0; i < n; i++) {
			int nxtState, nxtCost;
			//이미 켜져있는것
			if (bit_check(state, i)) {
				continue;
			}
			nxtState = bit_set(state, i);
			//j번째 발전소 이용해서 i켜기
			for (int j = 0; j < n; j++) {
				if (bit_check(state, j)) {
					nxtCost = dp[state] + w[j][i];
					dp[nxtState] = min(dp[nxtState], nxtCost);
				}
			}
		}
	}
	int ans = INF;
	for (int i = 0; i < (1 << n); i++) {
		int tmp = bit_count(i);
		if (tmp >= p) {
			ans = min(ans, dp[i]);
		}
	}
	if (ans == INF)cout << -1;
	else cout << ans;
	return 0;
}
~~~
