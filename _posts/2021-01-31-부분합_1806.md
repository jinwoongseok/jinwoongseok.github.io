[문제링크](https://www.acmicpc.net/problem/1759)
### [문제풀이]
투포인터 + 이분탐색 알고리즘을 사용했다. 부분합이 s보다 작다면 길이를 늘려가며 더해주고, 만약 s이상이라면 그 길이를 얻은 후, 길이를 줄여줌과 동시에 부분합에서 start 인덱스를 빼준다.  

### [소스코드]
~~~cpp
#include <iostream>
#include <algorithm>
using namespace std;
int n, s;
int A[100001];
const int INF = 987654321;
int ans=987654321;

void Solution() {
	int low = 0;
	int high = 0;
	int sum = A[0];
	while (true) {
		if (sum < s) {
			sum = sum + A[++high];
		}
		else {
			ans = min(high - low + 1, ans);
			sum = sum - A[low++];
		}
		if (high == n) {
			break;
		}
	}
}

int main() {
	cin >> n >> s;
	for (int i = 0; i < n; i++) {
		cin >> A[i];
	}
	Solution();
	if (ans == INF)cout << 0;
	else cout << ans;
	return 0;
}
~~~
